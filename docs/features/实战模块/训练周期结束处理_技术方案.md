class BattleState {
  // ... 现有字段 ...

  // === 新增字段 ===
  // 注意: initialStartIndex 已在现有代码中存在，默认值为 100

  // === 新增计算属性 ===
  int get trainingEndIndex {
    return initialStartIndex + trainingDays - 1;
  }

  bool get isTrainingComplete {
    return currentDayIndex >= trainingEndIndex;
  }

  int get currentTrainingDay {
    if (currentDayIndex < initialStartIndex) return 0;
    return (currentDayIndex - initialStartIndex + 1).clamp(1, trainingDays);
  }
}

## 3. 核心逻辑实现

### 3.1 BattleProvider.handleNextStep() 修改

**文件**: `lib/features/battle/providers/battle_provider.dart`

```dart
void handleNextStep() {
  // 复盘模式逻辑保持不变
  if (state.isReplayMode) {
    if (state.phase == TrainingPhase.opening) {
      setPhase(TrainingPhase.closing);
    } else {
      if (state.currentDayIndex < state.allKlineData.length - 1) {
        nextDay();
        setPhase(TrainingPhase.opening);
      }
    }
    return;
  }

  // === 新增: 训练周期结束判断 ===
  // 检查是否到达训练终点
  if (state.currentDayIndex >= state.trainingEndIndex) {
    // 训练已结束，显示结束对话框
    _showTrainingCompleteDialog();
    return;
  }

  // 正常训练流程
  if (state.phase == TrainingPhase.opening) {
    setPhase(TrainingPhase.closing);
  } else {
    if (state.currentDayIndex < state.allKlineData.length - 1) {
      nextDay();
      setPhase(TrainingPhase.opening);
    }
  }
}
```

**对应 AC**: AC-T001, AC-T002, AC-T003

### 3.2 BattleProvider.nextDay() 边界检查

```dart
void nextDay() {
  // === 新增: 训练终点边界检查 ===
  // 不能超过训练终点索引
  if (state.currentDayIndex >= state.trainingEndIndex) {
    return;
  }

  final newIndex = state.currentDayIndex + 1;

  state = state.copyWith(
    currentDayIndex: newIndex,
    phase: TrainingPhase.closing,
    visibleStartIndex: max(0, newIndex - state.visibleKlineCount + 1),
  );
}
```

**对应 AC**: AC-T001, AC-T002

### 3.3 训练记录保存方法

```dart
Future<int> _saveTrainingRecord() async {
  final dbService = DatabaseService.instance;
  final now = DateTime.now();

  // 计算最终资金
  final finalCapital = state.accountBalance + state.positionValue;
  final profitRate = (finalCapital - state.initialBalance) / state.initialBalance * 100;

  // 计算交易统计
  final buyPoints = state.tradePoints.where((p) => p.isBuy).toList();
  final sellPoints = state.tradePoints.where((p) => !p.isBuy).toList();
  final winCount = _calculateWinCount();
  final winRate = state.tradePoints.isEmpty ? 0.0 : winCount / state.tradePoints.length * 100;

  // 获取训练结束日期
  final endDate = state.allKlineData.isNotEmpty
      ? state.allKlineData[state.trainingEndIndex].dateTime
      : now;

  // 创建训练会话记录
  final sessionId = await dbService.trainingDao.createSession(
    TrainingSessionsCompanion.insert(
      symbol: state.currentSymbol,
      marketCode: state.currentMarketCode,
      startDate: state.trainingStartDate ?? endDate.subtract(Duration(days: state.trainingDays)),
      endDate: endDate,
      initialCapital: state.initialBalance,
      finalCapital: finalCapital,
      totalProfit: state.totalProfitLoss,
      tradeCount: state.tradePoints.length,
      winCount: winCount,
      winRate: winRate,
      createdAt: now,
    ),
  );

  print('✅ [BattleProvider] 训练记录已保存: sessionId=$sessionId');

  return sessionId;
}

int _calculateWinCount() {
  final buyPoints = state.tradePoints.where((p) => p.isBuy).toList();
  final sellPoints = state.tradePoints.where((p) => !p.isBuy).toList();

  if (sellPoints.isEmpty) return 0;

  int winCount = 0;
  int buyIndex = 0;

  for (final sell in sellPoints) {
    if (buyIndex < buyPoints.length) {
      final buy = buyPoints[buyIndex];
      if (sell.price > buy.price) {
        winCount++;
      }
      buyIndex++;
    }
  }

  return winCount;
}
```

### 4.2 边界处理增强

```dart
bool slideLeft() {
  // 不能滑动到训练起点之前
  if (state.visibleStartIndex <= state.initialStartIndex - state.visibleKlineCount) {
    final shouldAlert = _shouldShowBoundaryAlert(state.lastLeftBoundaryTime);
    if (shouldAlert) {
      state = state.copyWith(
        visibleStartIndex: state.initialStartIndex - state.visibleKlineCount,
        lastLeftBoundaryTime: DateTime.now(),
      );
    } else {
      state = state.copyWith(
        visibleStartIndex: state.initialStartIndex - state.visibleKlineCount,
      );
    }
    return shouldAlert;
  }

  state = state.copyWith(
    visibleStartIndex: state.visibleStartIndex - BattleConfig.slideStepCount,
  );
  return false;
}

bool slideRight() {
  // 不能滑动到训练终点之后
  final maxStart = (state.trainingEndIndex + 1 - state.visibleKlineCount)
      .clamp(0, state.trainingEndIndex + 1);

  if (state.visibleStartIndex >= maxStart) {
    final shouldAlert = _shouldShowBoundaryAlert(state.lastRightBoundaryTime);
    if (shouldAlert) {
      state = state.copyWith(
        visibleStartIndex: maxStart,
        lastRightBoundaryTime: DateTime.now(),
      );
    } else {
      state = state.copyWith(visibleStartIndex: maxStart);
    }
    return shouldAlert;
  }

  state = state.copyWith(
    visibleStartIndex: state.visibleStartIndex + BattleConfig.slideStepCount,
  );
  return false;
}
```

## 5. 数据展示范围控制

### 5.1 displayKlineData 修改

```dart
List<KlineData> get displayKlineData {
  if (state.allKlineData.isEmpty) return [];

  // 起点：不能早于数据开头
  // 终点：不能晚于训练终点
  final maxEndIndex = (state.trainingEndIndex + 1).clamp(0, state.allKlineData.length);

  final endIndex = (state.currentDayIndex + 1).clamp(0, maxEndIndex);
  final maxStart = (endIndex - state.visibleKlineCount).clamp(0, endIndex);
  final startIndex = state.visibleStartIndex.clamp(0, maxStart);

  return state.allKlineData
      .skip(startIndex)
      .take(min(state.visibleKlineCount, endIndex - startIndex))
      .map((e) => KlineData(
            date: e.dateTime,
            open: e.open,
            high: e.high,
            low: e.low,
            close: e.close,
            volume: e.volume,
          ))
      .toList();
}
```

## 6. 异常处理

### 6.1 边界情况处理

| 场景 | 处理方式 |
|------|---------|
| 训练天数小于预加载天数 | 训练起点索引 = 预加载天数 + 指标前置天数，固定不变 |
| 数据库数据不足 | 自动切换到数据充足的股票（已有逻辑） |
| 训练记录保存失败 | 仅在日志中输出错误，不影响用户操作 |
| 用户快速点击下一步 | 使用 debounce 防止重复触发 |

| 训练记录保存失败 | 低 | 中 | 仅记录日志，不阻塞用户操作 |
