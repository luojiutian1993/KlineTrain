class CurrentUserNotifier extends _$CurrentUserNotifier {

  Future<void> clearUser() async { state = null; }
}

Future<bool> silentLogin() async {
  final user = await UserRepository().getUserByPhone('13071103531');
  if (user != null) {
    state = true;
    ref.read(currentUserNotifierProvider.notifier).setUser(user);
    return true;
  }
  return false;
}

```dart
// lib/features/main/main_tab_page.dart
class _MainTabPageState extends ConsumerState<MainTabPage> {
  bool _initialized = false;
  
  @override
  void initState() {
    super.initState();
    WidgetsBinding.instance.addPostFrameCallback((_) {
      _initAuth();
    });
  }
  
  Future<void> _initAuth() async {
    if (_initialized) return;
    _initialized = true;
    
    final authState = ref.read(authStateProvider);
    if (!authState) {
      final success = await ref.read(authStateProvider.notifier).silentLogin();
      if (!success && mounted) {
        context.go(AppRoutes.login);
      }
    }
  }
}
```
