## 1. 单元测试

### 1.1 TimeRangeService 测试

**文件位置**：`test/data/services/time_range_service_test.dart`

```dart
void main() {
  late TimeRangeService service;
  late MockStockFilterDao mockDao;

  setUp(() {
    mockDao = MockStockFilterDao();
    service = TimeRangeService(mockDao);
  });

  group('TimeRangeService', () {
    const minDate = DateTime(2000, 1, 1);
    const maxDate = DateTime(2026, 5, 15);

    setUp(() {
      when(mockDao.getKlineDateRange()).thenAnswer(
        (_) async => (minDate, maxDate),
      );
    });

    test('generateRandomTrainingStartDate returns valid date within range', () async {
      final date = await service.generateRandomTrainingStartDate();

      // 验证日期不为空
      expect(date, isNotNull);

      // 验证日期在有效范围内（留出300天历史缓冲和150天训练周期）
      final earliestValidDate = minDate.add(const Duration(days: 300));
      final latestValidDate = maxDate.subtract(const Duration(days: 150));

      expect(date.isAfter(earliestValidDate), isTrue);
      expect(date.isBefore(latestValidDate), isTrue);
    });

    test('generateRandomTrainingStartDate throws when database has insufficient data', () async {
      // 设置数据库只有少量数据
      when(mockDao.getKlineDateRange()).thenAnswer(
        (_) async => (DateTime(2026, 1, 1), DateTime(2026, 2, 1)),
      );

      await expectLater(
        service.generateRandomTrainingStartDate(),
        throwsA(isA<Exception>()),
      );
    });

    test('generateRandomTrainingStartDate respects custom training days', () async {
      final date = await service.generateRandomTrainingStartDate(trainingDays: 60);

      final latestValidDate = maxDate.subtract(const Duration(days: 60));
      expect(date.isBefore(latestValidDate), isTrue);
    });
  });
}
```

```dart
        condition: StockFilterCondition.allTimeHigh,
        trainingStartDate: trainingStartDate,
```

```dart
void main() {
  late TimeRangeService service;
  late MockStockFilterDao mockDao;

  setUp(() {
    mockDao = MockStockFilterDao();
    service = TimeRangeService(mockDao);
  });

  group('TimeRangeService - Fallback Mechanism', () {
    test('uses default date range when database is empty', () async {
      // 模拟空表
      when(mockDao.getKlineDateRange()).thenAnswer(
        (_) async => (DateTime(2000, 1, 1), DateTime.now()),
      );

      // 执行
      final date = await service.generateRandomTrainingStartDate();

      // 验证正常返回，不抛异常
      expect(date, isNotNull);
      expect(date.isAfter(DateTime(2000, 1, 1)), isTrue);
      expect(date.isBefore(DateTime.now()), isTrue);
    });

    test('uses default date range when database has insufficient data', () async {
      // 模拟数据不足（只有30天数据，需要至少450天）
      when(mockDao.getKlineDateRange()).thenAnswer(
        (_) async => (DateTime(2026, 4, 1), DateTime(2026, 5, 1)),
      );

      // 执行
      final date = await service.generateRandomTrainingStartDate();

      // 验证使用默认日期范围，不抛异常
      expect(date, isNotNull);
      // 应该在历史5年范围内
      final fiveYearsAgo = DateTime.now().subtract(const Duration(days: 365 * 5));
      expect(date.isAfter(fiveYearsAgo), isTrue);
    });

    test('falls back to default when exception occurs', () async {
      // 模拟抛出异常
      when(mockDao.getKlineDateRange()).thenThrow(
        Exception('Database error'),
      );

      // 执行
      final date = await service.generateRandomTrainingStartDate();

      // 验证捕获异常并返回默认值
      expect(date, isNotNull);
    });

    test('uses available data when partially sufficient', () async {
      // 模拟有200天数据（够训练150天，但不够300天缓冲）
      when(mockDao.getKlineDateRange()).thenAnswer(
        (_) async => (
          DateTime(2026, 1, 1),

        ),
      );

      // 执行
      final date = await service.generateRandomTrainingStartDate();

      // 验证使用可用数据范围
      expect(date.isAfter(DateTime(2026, 1, 1)), isTrue);
      expect(date.isBefore(DateTime(2026, 7, 20).subtract(const Duration(days: 150))), isTrue);
    });

    test('_generateDefaultTrainingDate generates valid date', () {
      // 直接测试私有方法（通过反射或提取为公共方法测试）
      final now = DateTime.now();
      final fiveYearsAgo = DateTime(now.year - 5, now.month, now.day);
```
