  List<double> _computeMA(List<double> data, int period) {
    final result = <double>[];
    for (int i = 0; i < data.length; i++) {
      if (i < period - 1) {
        result.add(0);
      } else {
        double sum = 0;
        for (int j = i - period + 1; j <= i; j++) {
          sum += data[j];
        }
        result.add(sum / period);
      }
    }
    return result;
  }
